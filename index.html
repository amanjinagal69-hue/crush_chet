uploadStatus.style.color = "#ffcc00"; } } async function removeBackground() { const fileInput = document.getElementById('imageInput'); const preview = document.getElementById('preview'); const loadingText = document.getElementById('loadingText'); formData.append("image_file", file); formData.append("size", "auto"); loadingText.style.display = "block"; preview.style.display = "none"; downloadBtn.style.display = "none"; const apiKey = "ACRaE6rfpYnKpeL62fU3j2ER"; } finally { loadingText.style.display = "none"; } } function downloadImage() { if(globalImageUrl !== "") { const a = document.createElement('a'); a.href = globalImageUrl; a.download = "Aman_AI_Edited.png";
